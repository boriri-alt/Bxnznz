	local eggs = {
		["Common Egg"] = {"Dog", "Golden Lab", "Bunny"},
		["Mythical Egg"] = {"Grey Mouse", "Brown Mouse", "Squirrel", "Red Giant Ant"},
		["Bug Egg"] = {"Snail", "Giant Ant", "Caterpillar", "Praying Mantis"},
		["Anti-Bee Egg"] = {"Disco Bee", "Butterfly", "Moth", "Tarantula Hawk", "Wasp"},

		["Primal Egg"] = {"Parasaurolophus", "Iguanodon", "Pachycephalosaurus", "Dilophosaurus", "Ankylosaurus", "Spinosaurus"}
	}

	-- Function to generate a random pet from egg name
	local function getRandomPet(eggName)
		local petList = eggs[eggName]
		if petList then
			return petList[math.random(1, #petList)]
		else
			return "Unknown"
		end
	end

	-- Create ESP GUI on top of an egg part
	local function createESP(eggPart, eggName)
		if eggPart:FindFirstChild("EggESP") then return end

		local billboard = Instance.new("BillboardGui")
		billboard.Name = "EggESP"
		billboard.AlwaysOnTop = true
		billboard.Size = UDim2.new(0, 200, 0, 60)
		billboard.StudsOffset = Vector3.new(0, 3, 0)
		billboard.Parent = eggPart

		local label = Instance.new("TextLabel")
		label.Name = "PredictionLabel"
		label.Size = UDim2.new(1, 0, 1, 0)
		label.BackgroundTransparency = 1
		label.TextColor3 = Color3.fromRGB(255, 255, 0)
		label.TextStrokeTransparency = 0.4

	end

	-- Update ESP label with new random pet
	local function updatePrediction(eggPart, eggName)
		local esp = eggPart:FindFirstChild("EggESP")
		if esp and esp:FindFirstChild("PredictionLabel") then
			local pet = getRandomPet(eggName)
			esp.PredictionLabel.Text = "ü•ö " .. eggName .. "\nüîÆ " .. pet
		end
	end

	-- Scan for eggs in workspace and update ESP
	local function scanAndPredict()
		for _, obj in ipairs(workspace:GetDescendants()) do
			if obj:IsA("BasePart") and eggs[obj.Name] then
				createESP(obj, obj.Name)
				updatePrediction(obj, obj.Name)
			end
		end
	end

	-- Update prediction every 5 seconds
	while true do
		scanAndPredict()
		task.wait(5)
	end
    end
